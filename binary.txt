#include <iostream>
#include<algorithm>
#include <stdio.h>

using namespace std;
int arr[1000000];
int index;
void binarySearch(int l, int r, int x)
{
    while (l <= r) {
        int m = (l+r)/2;



        // If x greater, ignore left half
         if (arr[m] < x)
         {
            l = m + 1;
         }

        // If x is smaller, ignore right half
        else  if (arr[m] >= x)
        {
            if(index==-1)
            {

                if(arr[m]==x)
                {

                    index=m;
                }
            }
            else if(arr[m]==x)
            {

                index=min(index,m);
            }
            r = m - 1;
        }

    }

    // if we reach here, then element was
    // not present
}

int main()
{
int x = 0;
int n,m;
   cin>>n>>m;

   for(int i=0;i<n;i++)
   {
      scanf("%d",&arr[i]);
   }
   sort(arr,arr+n);
   for(int j=0;j<m;j++)
   {
      scanf("%d",&x);
      index=-1;
      binarySearch(0,n,x);
      printf("%d\n",index);



    x=0;

   }


    return 0;
}
