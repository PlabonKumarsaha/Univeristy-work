#include<stdio.h>
#include<math.h>
int f[100];
int main()
{
int low=0,value,co=0,a[100];
scanf("%d",&value);
int ma=0;
for(int i=0;i<value;i++)
{
    scanf("%d",&a[i]);
   if(f[a[i]]==0)
   {
 
       f[a[i]]++;
       co++;
       ma=ma>co?ma:co;
   }
   while(a[low]!=a[i])
   {
   low=low+1;
   co--;
   f[a[low]]--;
   }
low++;
   }
 
printf("%d",ma);
 
return 0
;}
